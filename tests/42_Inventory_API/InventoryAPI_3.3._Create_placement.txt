#1=all
#3=Mobile Web
#4=App

#2 = Disabled (return error response)

#'7', 'App Store URL (exchange agnostic)', '2'
#'8', 'Site Domain (exchange agnostic)', '3'

#'2', 'Site ID (exchange specific)', '4'
#'3', 'App ID (exchange specific)', '4'
